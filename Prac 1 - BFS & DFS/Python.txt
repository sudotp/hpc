from collections import deque
from concurrent.futures import ThreadPoolExecutor
import time

class Graph:
    def __init__(self, v):
        self.adj = {i: [] for i in range(v)}

    def add_edge(self, u, v):
        self.adj[u].append(v)
        self.adj[v].append(u)

    def parallel_bfs(self, start):
        visited = set([start])
        queue = deque([start])
        result = []
        while queue:
            node = queue.popleft()
            result.append(node)
            def try_visit(nb):
                if nb not in visited:
                    visited.add(nb)
                    queue.append(nb)
            with ThreadPoolExecutor() as ex:
                ex.map(try_visit, self.adj[node])
        return result

    def parallel_dfs(self, start):
        visited = set()
        stack = [start]
        result = []
        while stack:
            node = stack.pop()
            if node not in visited:
                visited.add(node)
                result.append(node)
                def try_push(nb):
                    if nb not in visited:
                        stack.append(nb)
                with ThreadPoolExecutor() as ex:
                    ex.map(try_push, self.adj[node])
        return result

g = Graph(7)
g.add_edge(0,1); g.add_edge(0,2)
g.add_edge(1,3); g.add_edge(1,4)
g.add_edge(2,5); g.add_edge(2,6)

t1 = time.time()
print("BFS:", g.parallel_bfs(0))
print(f"BFS time: {time.time()-t1:.6f}s")

t2 = time.time()
print("DFS:", g.parallel_dfs(0))
print(f"DFS time: {time.time()-t2:.6f}s")
