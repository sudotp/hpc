#include <iostream>
#include <vector>
#include <climits>
#include <omp.h>
using namespace std;

int main() {
    vector<int> arr = {3, 1, 7, 0, 9, 4, 6, 2, 8, 5};
    int n = arr.size();
    int minVal = INT_MAX, maxVal = INT_MIN, sum = 0;

    double t1 = omp_get_wtime();

    #pragma omp parallel for reduction(min:minVal) reduction(max:maxVal) reduction(+:sum)
    for (int i = 0; i < n; i++) {
        if (arr[i] < minVal) minVal = arr[i];
        if (arr[i] > maxVal) maxVal = arr[i];
        sum += arr[i];
    }

    double avg = (double)sum / n;
    double elapsed = omp_get_wtime() - t1;

    cout << "Array: ";
    for (int x : arr) cout << x << " ";
    cout << "\n\nMin:     " << minVal
         << "\nMax:     " << maxVal
         << "\nSum:     " << sum
         << "\nAverage: " << avg
         << "\n\nTime: "  << elapsed << "s\n";

    return 0;
}
